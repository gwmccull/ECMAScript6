Arrows =>

  • Shorthand for writing a function.
  • Similar to C#, Java 8 and CoffeeScript.
  • Supports both expression and statement bodies.
  • Shares the same this as the surrounding code.
var simple = a => a > 15 ? 15 : a;
simple(16); // 15
simple(10); // 10

var complex = (a, b) => {
  if (a > b) {
    return a;
  } else {
    return b;
  }
}

Let + Const

  • Block scoped variables.
  • let is the new var
  • const creates variables that can't be re-assigned.
function f() {
  {
    let x;
    {
      // okay, block scoped name
      const x = "sneaky";
      // error, const
      x = "foo";
    }
    // error, already declared in block
    let x = "inner";
  }
}

Promises

  • Allows for asynchronous calls.
  • Data is unknown so we have the promise of data.
p = new Promise( function() {} )
p.resolve( value )                      // resolve a promise with success
p.reject( reason )                      // resolve a promise with an error
Promise.all( [array] )                  // satisfied when all promises return
Promise.race( [array] )                 // satisfied when any promise returns
p.then( onFulfilled, onRejected )       // success/failure handler. May be chained
p.catch( onRejected )                   // global error handler. May be chained
